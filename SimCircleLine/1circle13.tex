\question Find the points of intersection between the circle \( x^2 + y^2 = 13 \) and each of the following lines.


\begin{multicols}{3}
\begin{parts}
    % Horizontal and vertical lines
    \part  \( x = -2 \)  \ms % Points used: (-2, -3) and (-2, 3)
    \part  \( x = 2 \) \ms % Points used: (2, -3) and (2, 3)
    \part  \( y = -3 \) \ms % Points used: (-2, -3) and (2, -3)
    \part  \( y = 3 \) \ms % Points used: (-2, 3) and (2, 3)

    % Skewed lines with slopes ±1/5 and ±5
    \part  \( x - 5y = -13 \) \ms % Points used: (-3, 2) and (2, 3), slope 1/5
    \part  \( x + 5y = 13 \) \ms % Points used: (3, 2) and (-2, 3), slope -1/5
    \part  \( 5x + y = -13 \) \ms % Points used: (-3, 2) and (-2, -3), slope 5
    \part  \( 5x + y = 13 \) \ms % Points used: (3, -2) and (2, 3), slope -5

    % Additional skewed lines
    \part  \( x - 5y = 13 \) \ms % Points used: (3, -2) and (-2, -3), slope 1/5
    \part  \( x + 5y = -13 \) \ms % Points used: (-3, -2) and (2, -3), slope -1/5
    \part  \( 5x + y = 13 \) \ms % Points used: (3, 2) and (2, -3), slope 5
    \part  \( 5x + y = -13 \) \ms % Points used: (-3, -2) and (-2, 3), slope -5
\end{parts}
\end{multicols}
