\documentclass{article}
\usepackage{tikz}
\begin{document}

% Define the line segment command
\newcommand{\linesegment}[6]{%
\begin{tikzpicture}
    \draw[step=1cm, gray, very thin] (0,0) grid (#1,#2);
    \draw[black, thick] (#3,#4) -- (#5,#6);
    \filldraw [black] (#3,#4) circle (2pt);
    \filldraw [black] (#5,#6) circle (2pt);
\end{tikzpicture}
}

% Example of using the command
\linesegment{11}{11}{1}{2}{4}{5}

\end{document}
