
\documentclass{article}
\usepackage{tikz}
\begin{document}

% Define the lines command
\newcommand{\lines}[2]{%
\begin{tikzpicture}
    \draw[step=1cm, gray, very thin] (0,0) grid (3,3);
    \draw[black, thick] (0,#1) -- (3,#2); % Line segment from (0,y1) to (3,y2)
    \filldraw [black] (0,#1) circle (2pt); % Endpoint at (0,y1)
    \filldraw [black] (3,#2) circle (2pt); % Endpoint at (3,y2)
\end{tikzpicture}
}

% Example of using the command
\lines{0}{3}

\end{document}
