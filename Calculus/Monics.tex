\question Recall and understand when to apply the following factoring and sign patterns:

\begin{multicols}{4}
\begin{parts}
    % Positive + Positive factors
    \part $(x + \hspace{11} )(x + \hspace{11} )$
    % Negative + Negative factors
    \part $(x - \hspace{11} )(x - \hspace{11} )$
    % Mixed signs: large positive, small negative
    \part $(x + \text{big})(x - \text{small})$
    % Mixed signs: large negative, small positive
    \part $(x - \text{big})(x + \text{small})$
    % Sign patterns for general quadratic
    \part Sign patterns for $Ax^2 + Bx + C$
    \begin{itemize}
        \item If $A > 0$, $B > 0$, and $C > 0$: $(x + \hspace{11})(x + \hspace{11})$
        \item If $A > 0$, $B < 0$, and $C > 0$: $(x - \hspace{11})(x - \hspace{11})$
        \item If $C < 0$: Mixed signs $(x + \hspace{11})(x - \hspace{11})$
    \end{itemize}
    % Simple space for custom content if necessary
    \part 
    % Space for turning points
    \part Turning points and their signs
    \begin{itemize}
        \item $x = -\frac{B}{2A}$
        \item Nature of turning point depends on sign of $A$
    \end{itemize}
    % Second derivative discussion
    \part Second derivative and concavity
    \begin{itemize}
        \item $f''(x) = 2A$
        \item If $A > 0$: Concave up (minimum)
        \item If $A < 0$: Concave down (maximum)
    \end{itemize}
\end{parts}
