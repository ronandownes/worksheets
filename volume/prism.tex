        % Vertical edges (dashed for the back)

        \draw[dashed] (1,-1) -- (0,0);  % bottom back vertical (dashed)

        % Shade the bottom face
        \fill[gray!20] (1,-1) -- (4,-1) -- (0,0) -- cycle;
