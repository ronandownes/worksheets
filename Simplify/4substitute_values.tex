    \part $2x + y$ \hs
    \part $3y + 2z$ \hs
    \part $5x + z$ \hs
    \part $4x + 3z$ \hs

    \part $6y - 2z$ \hs
    \part $3x - 6y$ \hs
    \part $6x - 8z$ \hs
