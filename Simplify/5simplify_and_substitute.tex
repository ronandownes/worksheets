\section*{Simplify and Substitute}

% First, simplify and then substitute the values given for variables
\question First, simplify and find the value, given \( a = 5 \), \( b = 4 \), and \( c = 2 \).

\begin{multicols}{2}
\begin{parts}\Large
    \part $2a + 4b + a + 2b$ \hs
    \part $3a + 2c + 3c + a$ \hs
    \part $4b + c + 2b + c$ \hs
    \part $6a + b - 3a + 2b$ \hs
    \part $5c + 2a - c + 4b - a$ \hs
    \part $3a + b + 2c - a - b - 2a$ \hs
    \part $4c + 3b - c + b - a$ \hs
    \part $4a + 3b + 3c - 3b + a$ \hs
    \part $3a + 2b + 2c - a - 2b$ \hs
    \part $5b + c - 3b + 2a - c$ \hs
    \part $8a + 7b - 5c + a$ \hs
    \part $6a + 4b - 5a + 3c$ \hs
\end{parts}
\end{multicols} \ms
