
\section*{Simplifying and Evaluating Expressions}

% Simplify the expressions by combining like terms
\question Simplify by combining \textbf{like terms}.
\begin{multicols}{2}
\begin{parts}
    \part $a + a + a$ \is  % Q1: Answer: 3a
    \part $b + 2b + 3b$ \is % Q2: Answer: 6b
    \part $5c + 2c + c$ \is % Q3: Answer: 8c
    \part $4d + 3d + 7d$ \is % Q4: Answer: 14d
    \part $6e + 3e - 7e$ \is % Q5: Answer: 2e
    \part $3f + 8f - 9f$ \is % Q6: Answer: 2f
    \part $4x + 5x - 7x$ \is % Q7: Answer: 2x
    \part $8y - 3y - y$ \is % Q8: Answer: 4y
    \part $7z + 8z - 9z - 5z$ \is % Q9: Answer: z
    \part $3x - x + 4x - x$ \is % Q10: Answer: 5x
    \part $7j + y - 5y - j$ \is % Q11: Answer: 6j - 4y
    \part $2 + 6z + 2z - 9z$ \is % Q12: Answer: 2 - z
\end{parts}
\end{multicols} \is
