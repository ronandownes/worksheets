\newcommand{\randcoeff}{% Command to generate random coefficients
    \pgfmathtruncatemacro{\temp}{random(-13,13)}%
    \ifnum\temp=0 \randcoeff \else\temp\fi% Avoid zero coefficients
}

\newcommand{\formatcoeff}[1]{% Format coefficient, omit if 1 or -1
    \ifnum#1=1 x \else \ifnum#1=-1 -x \else #1x \fi \fi%
}

\newcommand{\randexp}{%
    \pgfmathtruncatemacro{\a}{\randcoeff}%
    \pgfmathtruncatemacro{\b}{\randcoeff}%
    \pgfmathtruncatemacro{\c}{\randcoeff}%
    \formatcoeff{\a}(\formatcoeff{\b} \pgfmathparse{sign(\c)==-1 ? '' : '+'}\pgfmathresult \c)%
}

\newcommand{\randquadexp}{% Example for quadratic expression
    \pgfmathtruncatemacro{\d}{\randcoeff}%
    \pgfmathtruncatemacro{\e}{\randcoeff}%
    \pgfmathtruncatemacro{\f}{\randcoeff}%
    \formatcoeff{\d}(\formatcoeff{\e}^2 \pgfmathparse{sign(\f)==-1 ? '' : '+'}\pgfmathresult \f)%
}
