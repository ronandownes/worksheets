\newcommand{\randcoef}{\pgfmathtruncatemacro{\temp}{random(1,5)}\temp}

% Define a command for generating perfect squares
\newcommand{\perfectsquare}{%
    \pgfmathtruncatemacro{\a}{random(1,5)} % Random coefficient for x
    \pgfmathtruncatemacro{\b}{random(1,5)} % Random coefficient for constant
    \edef\result{(\a x + \b)(\a x + \b)}%
    \result
}

% Define a command for generating the difference of squares
\newcommand{\differenceofsquares}{%
    \pgfmathtruncatemacro{\a}{random(1,5)} % Random coefficient for x
    \pgfmathtruncatemacro{\b}{random(1,5)} % Random coefficient for constant
    \edef\result{(\a x + \b)(\a x - \b)}%
    \result
}

% Define a command for generating trivial form (x + b)(x + b)
\newcommand{\trivialperfectsquare}{%
    \pgfmathtruncatemacro{\b}{random(1,5)} % Random coefficient
    \edef\result{(x + \b)(x + \b)}%
    \result
}

% Define a command for generating general form
\newcommand{\generalperfectsquare}{%
    \pgfmathtruncatemacro{\a}{random(1,5)} % Random coefficient for general form
    \pgfmathtruncatemacro{\b}{random(1,5)} % Random coefficient for general form
    \edef\result{(\a x + \b)(\a x + \b)}%
    \result
}
